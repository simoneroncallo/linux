# smartctl
lsblk -f
sudo smartctl -a /dev/sdX # Print SMART attributes

# btrfs (sudo)
btrfs scrub start / # Integrity check on mounted filesystem 
btrfs scrub status / # Print status
btrfs scrub cancel /

# mount
OPT=nosuid,nodev,nofail,noauto,x-gvfs-show,uid=1000,gid=1000,fmask=133,dmask=022 # Mount HDD
sudo mount -o "$OPT" /dev/sdX /mnt/point
