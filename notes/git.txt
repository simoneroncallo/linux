# git
git config --global user.name "username" # Username
git config --global user.email "you@example.com" # Email

git clone https://github.com//<username>/<name>.git # Clone public repository
cd <name>

git status
git add . # Stage
git diff --cached # Diff
git commit -m "Message" # Commit
git push origin main # Push

git fetch origin 
git pull origin main
git status
